Manuál – základní příkazy

cd <složka> – přejde do zadané složky

ls – vypíše obsah aktuální složky

cat <soubor> – vypíše obsah souboru

echo "text" > <soubor> – zapíše text do souboru (přepíše obsah)

echo "text" >> <soubor> – přidá text na konec souboru

git pull – stáhne nejnovější změny z repozitáře

git commit -a -m "zpráva" – uloží změny s popisem

git push – odešle změny na GitHub
